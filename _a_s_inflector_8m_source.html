223 - (void)addIrregularWithSingular:(NSString *)singular plural:(NSString *)plural
224 {
225  [uncountables removeObject:singular];
226  [uncountables removeObject:plural];
227  // Assume that the singular and plural string arguments are not empty. Both
228  // must have lengths equal to or greater than one.
229  NSString *s_1 = [singular substringToIndex:1];
230  NSString *p_1 = [plural substringToIndex:1];
231  NSString *s1_ = [singular substringFromIndex:1];
232  NSString *p1_ = [plural substringFromIndex:1];
233  if ([[s_1 uppercaseString] isEqualToString:[p_1 uppercaseString]])
234  {
235  [self addPluralRegularExpressionRule:[NSString stringWithFormat:@"(%@)%@$", s_1, s1_] options:NSRegularExpressionCaseInsensitive replacement:[@"$1" stringByAppendingString:p1_]];
236  [self addPluralRegularExpressionRule:[NSString stringWithFormat:@"(%@)%@$", p_1, p1_] options:NSRegularExpressionCaseInsensitive replacement:[@"$1" stringByAppendingString:p1_]];

318 - (NSString *)humanize:(NSString *)lowerCaseAndUnderscoredWord
319 {
320  NSMutableString *result = [ASInflectorApplyRulesAndReplacements(humans, lowerCaseAndUnderscoredWord) mutableCopy];
321 
322  // Strip any trailing "_id" string. The method could implement this without
323  // using regular expressions. Such might prove more efficient. Rails uses
324  // regular expressions however and since this framework aims to mimic Rails,
325  // use a regular expression too.
326  [[NSRegularExpression regularExpressionWithPattern:@"_id$" options:0 error:NULL] replaceMatchesInString:result options:0 range:NSMakeRange(0, [result length]) withTemplate:@""];
327 
328  [result replaceOccurrencesOfString:@"_" withString:@" " options:0 range:NSMakeRange(0, [result length])];
329 
330  // Match sequences of letters and numbers case-insensitively. Does the match
331  // correspond to an acronym? If it does, replace the acronym with its
332  // case-sensitive equivalent. Note, some acronyms use both upper and lower
333  // cases, e.g. RESTful. The acronyms dictionary keys contain acronyms in
334  // lower case. The argument is a lower-case and underscored word.
335  ASReplaceMatchesInStringUsingBlock(@"([a-z\\d]*)", NSRegularExpressionCaseInsensitive, result, ^NSString *(NSArray *results) {
336  NSString *result0 = [results objectAtIndex:0], *acronym = [acronyms objectForKey:result0];
337  return acronym ? acronym : [result0 lowercaseString];
338  });
339 
340  return ASStringByReplacingMatchesInStringUsingBlock(@"^\\w", 0, result, ^NSString *(NSArray *results) {
341  return [[results objectAtIndex:0] uppercaseString];
342  });
343 }
