139 - (void)testRFC2822
140 {
141  NSDateComponents *dateComponents = [[NSDateComponents alloc] init];
142  [dateComponents setYear:1999];
143  [dateComponents setMonth:12];
144  [dateComponents setDay:31];
145  [dateComponents setHour:19];
146  [dateComponents setMinute:0];
147  [dateComponents setSecond:0];
148  NSCalendar *calendar = [[NSCalendar alloc] initWithCalendarIdentifier:NSGregorianCalendar];
149  [calendar setTimeZone:[NSTimeZone timeZoneWithName:@"EST"]];
150  NSDate *date = [calendar dateFromComponents:dateComponents];
151  // Note, ASRFC2822StringFromDate applies the current time zone when
152  // converting from a date. This may not be Greenwich Mean Time. Fixing the
153  // time zone occurs when the ActiveSupport date formatter constructs its
154  // dependent NSDateFormatter objects. The sub-formatters apply the default
155  // time zone.
156  NSDateFormatter *dateFormatter = [[NSDateFormatter alloc] init];
157  [dateFormatter setLocale:[[NSLocale alloc] initWithLocaleIdentifier:@"en_US_POSIX"]];
158  [dateFormatter setDateFormat:@"dd MMM yyyy HH:mm ZZZ"];
159  STAssertEqualObjects(ASRFC2822StringFromDate(date), [dateFormatter stringFromDate:date], nil);
160 }
