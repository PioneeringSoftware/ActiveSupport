101 - (void)testAcronyms
102 {
103  [[ASInflector defaultInflector] addAcronym:@"API"];
104  [[ASInflector defaultInflector] addAcronym:@"HTML"];
105  [[ASInflector defaultInflector] addAcronym:@"HTTP"];
106  [[ASInflector defaultInflector] addAcronym:@"SSL"];
107  [[ASInflector defaultInflector] addAcronym:@"RESTful"];
108  [[ASInflector defaultInflector] addAcronym:@"W3C"];
109  static struct
110  {
111  NSString *const __unsafe_unretained camel;
112  NSString *const __unsafe_unretained under;
113  NSString *const __unsafe_unretained human;
114  NSString *const __unsafe_unretained title;
115  }
116  camelUnderHumanTitle[] =
117  {
118  { @"API", @"api", @"API", @"API" },
119  { @"APIController", @"api_controller", @"API controller", @"API Controller" },
120  { @"Nokogiri::HTML", @"nokogiri/html", @"Nokogiri/HTML", @"Nokogiri/HTML" },
121  { @"HTTPAPI", @"http_api", @"HTTP API", @"HTTP API" },
122  { @"HTTP::Get", @"http/get", @"HTTP/get", @"HTTP/Get" },
123  { @"SSLError", @"ssl_error", @"SSL error", @"SSL Error" },
124  { @"RESTful", @"restful", @"RESTful", @"RESTful" },
125  { @"RESTfulController", @"restful_controller", @"RESTful controller", @"RESTful Controller" },
126  { @"IHeartW3C", @"i_heart_w3c", @"I heart W3C", @"I Heart W3C" },
127  };
128  for (NSUInteger i = 0; i < sizeof(camelUnderHumanTitle)/sizeof(camelUnderHumanTitle[0]); i++)
129  {
130  STAssertEqualObjects(camelUnderHumanTitle[i].camel, [[ASInflector defaultInflector] camelize:camelUnderHumanTitle[i].under uppercaseFirstLetter:YES], nil);
131  STAssertEqualObjects(camelUnderHumanTitle[i].camel, [[ASInflector defaultInflector] camelize:camelUnderHumanTitle[i].camel uppercaseFirstLetter:YES], nil);
132  STAssertEqualObjects(camelUnderHumanTitle[i].under, [[ASInflector defaultInflector] underscore:camelUnderHumanTitle[i].under], nil);
133  STAssertEqualObjects(camelUnderHumanTitle[i].under, [[ASInflector defaultInflector] underscore:camelUnderHumanTitle[i].camel], nil);
134  STAssertEqualObjects(camelUnderHumanTitle[i].title, [[ASInflector defaultInflector] titleize:camelUnderHumanTitle[i].under], nil);
135  STAssertEqualObjects(camelUnderHumanTitle[i].title, [[ASInflector defaultInflector] titleize:camelUnderHumanTitle[i].camel], nil);
136  STAssertEqualObjects(camelUnderHumanTitle[i].human, [[ASInflector defaultInflector] humanize:camelUnderHumanTitle[i].under], nil);
137  }
138 }
